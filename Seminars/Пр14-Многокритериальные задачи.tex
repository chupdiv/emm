Характеристики КСА приведены в таблице:

    \begin{tabular}{|p{6cm}|c|c|c|c|c|c|c|}

        \hline

        40 &  30 & 40 & 60 & 45 & 25 & 55 \\
        \hline
        Срок ввода в эксплуатацию, мес. & 
        8 & 8 & 6 & 6 & 7 & 8 & 6 \\
        \hline
        Срок гарантийного обслуживания, лет &
        4 & 4 & 5 & 7 & 4 & 4 & 5 \\

        \hline
    \end{tabular}
    \par}
